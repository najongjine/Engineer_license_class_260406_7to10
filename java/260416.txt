package test2604;

import java.util.ArrayList;
import java.util.List;

class A{
	A(int a){
		System.out.println("A탄생");
	}
	void end() {
		System.out.println("끝");
	}
}
class B extends A{
	/* 자바 상속관계에서 기본으로 부모의 기본 생성자를 호출하려 한다
	 * 그런데, A() 이런 기본 생성자가 없으면, 우리가 수동으로 부모 생성자를 호출해야 한다.
	 * */
	B(){
		super(1);
		System.out.println("B탄생");
	}
}

public class Test2604 {
	public static void main(String[] args) {
		B b=new B();
	}
}

package test2604;

import java.util.ArrayList;
import java.util.List;
/* abstract (추상클래스).
 * 인터페이스 나오기 전에 쓰던거
 * 
 * 꼭 만들다 만것처럼 생김
 * 몸체가 없는 abstract 함수도 선언할수 있고,
 * 몸체가 있는 함수도 선언할수 있다.
 * 데이터도 선언할수 있다.
 * 
 * 단, abstract class 단독으로는 new 할수 없다.
 * 꼭 상속받아서 사용 해야 한다.
 * */
abstract class Shape{
	int a=2;
	abstract void draw();
	void color(String color) {
		System.out.println(color + "색이에요");
	}
}
class Circle extends Shape{
	void draw() {
		System.out.println("동그라미");
	}
}


public class Test2604 {
	public static void main(String[] args) {
		Shape shape=new Circle();
	}
}

package test2604;

import java.util.ArrayList;
import java.util.List;
/*
 * 인터페이스 - 청사진. 클래스의 설계도
 * 디자인 패턴을 위해서 나온놈
 * 요놈을 좀 어렵게 사용하면 다형성도 된다.
 * */
interface Animal{
	void move();
	void eat();
}
interface Animal2{
	void move();
	void eat();
}
class OldDog{
	int u=3;
}
/*
 * 인터페이스는 여러개 상속 받을수 있다.
 * 인퍼테이스에 선언된 method는 반드시 상세구현 해줘야 된다
 * */
class Dog extends OldDog implements Animal,Animal2{
	int a=2;
	public void move() {
		System.out.println("달려요");
	}
	public void eat() {
		System.out.println("먹어요");
	}
}

public class Test2604 {
	public static void main(String[] args) {
		Animal dog1=new Dog();
		Animal dog2=new Dog();
		dog1.eat();
		dog2.move();
	}
}

package test2604;

import java.util.ArrayList;
import java.util.List;

interface Vehicle {
    void run();
}

class Bus implements Vehicle {
    public void run() {
        System.out.println("버스출발 ");
    }
}

class Taxi implements Vehicle {
    public void run() {
        System.out.println("택시출발 ");
    }
}

public class Test2604 {
	public static void main(String[] args) {
		 Vehicle[] arr = { new Bus(), new Taxi() };
        for(Vehicle v : arr) {
            v.run();
        }
	}
}
