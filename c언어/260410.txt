#include <stdio.h>

/*
void : 없다
myf : 내가 짓고싶은 이름(함수 이름)
() : 매개변수. parameter
{ } : body. 몸체
*/
void myf() {
    printf("i'm myf");
}

int myf2() {
    int i = 2;
    return i; // 2 퉤 뱉고 기계 꺼짐
}
int myf3(int i) {
    return i + 1;
}
int myf4(int i,char a) {
    return i + a;
}
int main() {
    myf();
    int r1=myf2();
    r1 = myf3(1);
    r1=myf4(1,'A');
    return 0;
}

#include <stdio.h>

int str_length(char str1[]) {
    int i = 0;
    for (;1;i++) {
        printf("%c \n",str1[i]);
        if (str1[i] == '\0') break;
    }
    return i;
}
int main() {
    char str[] = "hello";
    int length=str_length(str);
    printf("length: %d \n",length);
    return 0;
}
