#include <stdio.h>


void insertionSort(int a[], int n) {
    int i, j, key;
    for (i = 1; i < n; i++) {
        key = a[i];
        for (j = i - 1; j >= 0 && a[j] > key; j--) {
            a[j + 1] = a[j];
        }
        a[j + 1] = key;
    }
}

int main() {
    int a[5] = { 8, 5, 6, 2, 4 };
    int i;

    insertionSort(a, 5);

    for (i = 0; i < 5; i++) {
        printf("%d ", a[i]);
    }

    return 0;
}

#include <stdio.h>


void bubbleSort(int a[], int n) {
    int i, j, temp;

    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (a[j] > a[j + 1]) {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }
}

int main() {
    int a[] = { 5, 3, 8, 1, 2 };
    int n = 5;
    int i;

    bubbleSort(a, n);

    for (i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }

    return 0;
}
