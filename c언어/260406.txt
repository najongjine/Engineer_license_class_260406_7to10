#include <stdio.h>

/*
* c, java는 무조건 main() 함수부터 찾아서
* 실행해요
*/
/*이걸 주석이라고(comment) 해요
* 컴퓨터가 여기 주석 부분은 무시해요.
* 그래서 에러가 안나는 거에요
*/
//이것도 주석이라고 해요
int main() {
	/*
	* 변수 := 메모리 주소 어딘가에 별명 붙여
	* 크기는 int만큼 잡아
	* int는 숫자만 저장할수 있어요
	*/
	int a;

	/* 문맥 
	* =(대입연산)
	* 왼쪽에 있으면 저장, 오른쪽에 있으면 꺼내오기
	*/
	a = 1;
	a = 'a';
	printf("a: %d \n",a);
	

	return 0;
}
