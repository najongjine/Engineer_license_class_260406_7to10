#include <stdio.h>

/*
* c, java는 무조건 main() 함수부터 찾아서
* 실행해요
*/
/*이걸 주석이라고(comment) 해요
* 컴퓨터가 여기 주석 부분은 무시해요.
* 그래서 에러가 안나는 거에요
*/
//이것도 주석이라고 해요
int main() {
	/*
	* 변수 := 메모리 주소 어딘가에 별명 붙여
	* 크기는 int만큼 잡아
	* int는 숫자만 저장할수 있어요
	*/
	int a;

	/* 문맥 
	* =(대입연산)
	* 왼쪽에 있으면 저장, 오른쪽에 있으면 꺼내오기
	*/
	a = 1;
	a = 'a';
	printf("a: %u \n",a);

	/*
	* c, java 는 같은 scope 안에 똑같은 이름의 변수를
	* 또 선언할수 없어요. 왜냐면 컴퓨터는
	* 똑같은 별명 2개가 있으면 구분을 못해요
	int a;
	a = 1;
	*/
	a = 1;
	int b = 1;
	int c = a + b;
	printf("c: %d \n", c);

	a = 1;
	b = 2;
	(float)a;// 성형수술
	printf("%d(a)/%d(b): %d \n", a, b, a / b);
	printf("%d(a)/%d(b): %f \n",a,b,(float)a/b);

	// 문자
	char c1 = 'a';
	printf("%c \n", c1);

	// 문자열
	char* c2 = "savdcbfg";
	char c3[100] = "dfdsgfdf";

	return 0;
}
