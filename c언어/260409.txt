#include <stdio.h>

int main() {
    // 반복문 := 조건이 만족되는 동안만 반복수행 한다.
    int i = 1;
    while (i) { 
        i++;
        if (i==3) {
            break;
        }
        printf("%d \n", i);
    }

    return 0;
}

int main() {
