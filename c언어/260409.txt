#include <stdio.h>

int main() {
    // 반복문 := 조건이 만족되는 동안만 반복수행 한다.
    int i = 1;
    while (i) { 
        i++;
        if (i==3) {
            break;
        }
        printf("%d \n", i);
    }

    return 0;
}

#include <stdio.h>

int main() {
    // 반복문 := 조건이 만족되는 동안만 반복수행 한다.
    int i = 3;
    
    switch (i) {
        case 3:
            printf("i= 3 \n");
            break;
        case 2:
            printf("i= 2 \n");
            break;
        default:
            printf("dunno");
    }

    return 0;
}

#include <stdio.h>

int main() {
    int i = 0;
    
    do { // do-while := 일단 한번은 무조건 실행하고 조건 확인
        printf("i:%d \n",i);
        i++;
    } while (i < 2);

    return 0;
}
