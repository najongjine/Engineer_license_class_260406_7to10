#include <stdio.h>

int main() {
    putchar('c'); // 문자만 출력. 다음줄 개행 자동으로 안함.
    puts("puts...%d",10);//문장만 출력. 다음줄 개행 자동으로 함.
    printf("printf... %d \n",10); // 가장흔히 씀. 다음줄 개행 수동으로 해줘야함

    /*
    * malloc은 동적 메모리 할당
    * int *p;
      p = (int *)malloc(sizeof(int));
    */

   
    int test1;
    int* p;

    /* malloc: Heap 메모리 공간에 메모리 동적으로 할당해라. 
    그 Heap메모리에 할당한 주소를 p에 저장해라*/
    p = (int*)malloc(sizeof(int));   
    // p의 원산지 찾아가서 거기에 100을 저장해라
    *p = 100;

    printf("%d\n", *p);

    free(p);   // 사용 끝나면 해제. 이거 시험에선 의미 없는 코드

    int test[5]; // 밑에 예제는 이거랑 동급
    int* arr;
    int i;
    /* Heap 메모리 공간에 malloc으로 동적 공간 할당 하는데, int 사이즈만큼
    5개 연속으로 만들어라.
    이러면 배열이랑 완전 똑같음 */
    arr = (int*)malloc(sizeof(int) * 5);  // int 5개 공간 할당

    for (i = 0; i < 5; i++) {
        arr[i] = (i + 1) * 10;
    }

    for (i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }

    free(arr);

    return 0;
}

#include <stdio.h>

int main() {
    /* 호랑이 담배피던 시절 
    * int name[100];
    * char* phone[100];
    * int money[100];
    * 
    * name[0]; <- 페페
    * phone[0];
    * bubbleSort(money);
    * money[0];
    * 
    * 프로그래밍 초창기때는 이렇게 하면 어떤사람이 어떤데이터를 가지고 있는지
    * 보장이 안되요. 정렬 한번 때리면 그냥 초토화 되는거에요
    * 
    * 그래서 이런 제안을 해요.
    * 
    * 관련있는건 싸맵시다.
   struct Bank {
    char* name;
    int money;
    char* phone;
   };

   struct는 heap 메모리에 공간 만들고 시작주소 저장하는 포인터 방식이 아니라,
   그냥 main 메모리 같은곳에 걍 데이터 뭉탱이 싸매서 만든다.
   즉, 포인터 아니다
    */
    struct Bank {
        char* name;
        int money;
        char* phone;
    } ;
    
    struct Bank s1;
    s1.name = "pepe";
    s1.money = 5000;
    s1.phone = "010111";

    struct Bank s2; // s1 이랑 s2는 완전 다른 놈이에요
    s2.name = "speaky";
    s2.money = 6000;
    s2.phone = "010222";
    
    printf("s1.name : %s \n", s1.name);
    printf("s1.money : %d \n", s1.money);
    printf("s1.phone : %s \n", s1.phone);
    
    return 0;
}

#include <stdio.h>
#include <string.h>

typedef struct {
    char city[20];
    char phone[20];
} Address;

typedef struct {
    char name[20];
    int age;
    Address addr;   // 구조체 안에 구조체
} Person;

int main() {
    Person p1;

    strcpy(p1.name, "Kim");
    p1.age = 25;

    strcpy(p1.addr.city, "Seoul");
    strcpy(p1.addr.phone, "010-1234");

    printf("이름: %s\n", p1.name);
    printf("나이: %d\n", p1.age);
    printf("도시: %s\n", p1.addr.city);
    printf("전화: %s\n", p1.addr.phone);

    return 0;
}

#include <stdio.h>
#include <string.h>

typedef struct {
    char name[20];
    int age;
    
} Person;

int main() {
    Person p1 = {"pepe",20};
    Person* pp1 = &p1;
    printf("%d \n", p1.age);
    // struct을 포인터로 했을땐 -> 기호로 속성에 파고들어야 한다
    printf("%d \n", pp1->age);
   

    return 0;
}
