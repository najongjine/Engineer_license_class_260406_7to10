package test2604;

public class Test2604 {
	//                           stamp 결합도   자료 결합도
	public static void change(String[] data, String s){
		System.out.println("change data:"+data);
        data[0] = s; // main 에 있는 data 배열의 0번째 원소를 "B" 로 바꿔라
        s = "Z";
    }
	public static void main(String[] args) {
		String data[] = { "A" }; // "B"
        String s = "B";
        
        change(data, s);
        System.out.print(data[0] + s);
	}
}
