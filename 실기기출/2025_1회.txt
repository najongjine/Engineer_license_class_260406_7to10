#include <stdio.h>
//               66   65    67   68   69
//             { 'B', 'A', 'C', 'D' , 'E'};
char Data[5] = { 'B', 'A', 'D', 'E' };

char c;

int main() {
    int i; // 5
    int temp; // 
    int temp2;

    c = 'C'; // 67
    //              E(69)  -  A(65) = 4
    printf("%d\n", Data[3] - Data[1]);

    // i 0~ 4, 5에서 빠져와요
    for (i = 0; i < 5; ++i) {
        // Data[2] > 'C' -> 'D' > 'C' -> T
        // 여기서 i 가 2가 되요
        if (Data[i] > c)
            break;
    }

    temp = Data[i]; // t= Data[2] -> 'D' // 68
    Data[i] = c; // Data[2] = 'C'
    i++; // 3

    // i 3~4, 5에 빠져나옴
    for (; i < 5; ++i) {
        temp2 = Data[i]; // t2=Data[4] -> t2=''
        Data[i] = temp; // Data[4]='E'
        temp = temp2; // t=''
    }

    for (i = 0; i < 5; i++) {
        printf("%c", Data[i]);
    }
}

#include <stdio.h>
#include <stdlib.h>
/* 
* {
*  {9,5,2}
*  {7,4,1}
*  {8,3,6}
* }
*/
//                                3           3
void set(int** arr, int* data, int rows, int cols) {
    for (int i = 0; i < rows * cols; ++i) {// i 0~8, 9에서 빠져나옴
      //arr[0][0] = 9;
        arr[((i + 1) / rows) % rows][(i + 1) % cols] = data[i];
    }
}

int main() {
    int rows = 3, cols = 3, sum = 0;
    
    int data[] = { 5, 2, 7, 4, 1, 8, 3, 6, 9 };
    /*
    * arr = {
    *  {9,5,2}
    *  {7,4,1}
    *  {8,3,6}
    * }
*/
    int** arr;
    arr = (int**)malloc(sizeof(int*) * rows);
    for (int i = 0; i < cols; i++) {
        arr[i] = (int*)malloc(sizeof(int) * cols);
    }
    printf(" %d\n",1/3 );
    printf("arr[0]: %p \n", arr[0]);
    set(arr, data, rows, cols);
    // i 0~8, 9에서 빠져나와요
    // sum=13
    for (int i = 0; i < rows * cols; i++) {// i 8, 6
      //sum += 6;
        sum += arr[i / rows][i % cols] * (i % 2 == 0 ? 1 : -1);
    }

    for (int i = 0; i < rows; i++) {
        free(arr[i]);
    }
    free(arr);

    printf("%d", sum);
}

package test2604;

class Parent {
    static int total = 0; // 18
    int v = 1;
 
    public Parent() {
        total += (++v); //total+= 2 -> total=2
        show();    
    }
    // 상속관계 에서는 자식이랑 부모랑 똑같은 함수 있으면 자식꺼로 override
    public void show() {
        total += total;
    }
}
 
 
class Child extends Parent {
    int v = 10; // 13
 
    public Child() {
        v += 2;
        total += v++; // total= 6+12, v++
        show();
    }
 
    @Override // 계속 이것만 실행됨
    public void show() {
        total += total * 2; //t= 18 + 36
    }
}

public class Test2604 {
	public static void main(String[] args) {
		new Child();
        System.out.println(Parent.total);
	}
}
