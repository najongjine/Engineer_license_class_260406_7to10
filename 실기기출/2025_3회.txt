#include <stdio.h>

int main(void) {
    // c언어는 문자열 끝에 \0 를 컴파일때 자동으로 넣어요
    char str[] = "REPUBLICOFKOREA";
    int a = 0; // 15
    

    // c언어에서 문자열 길기 구하는 코드.
    while (str[a] != '\0')
        ++a;
    // str[13]
    putchar(str[a - 2]);
    return 0;
}

#include <stdio.h>

struct Node {
    struct Node* next;
    unsigned int x;
};

int main() {
    struct Node t1 = { 0, 5u };
    struct Node t2 = { 0, 7u };
    struct Node t3 = { 0, 11u };

    t3.next = &t2;
    t2.next = &t1;

    struct Node* curr = &t3;
    int sum = 0; // 125

    while (curr) {
        sum = sum * 3 + curr->x;
        curr = curr->next;
    }
    // sum = (125 ^ 42) + 100 -> (1111101 ^ 101010)    ^ := xor 연산
    sum = (sum ^ 42u) + 100u;

    printf("%u\n", sum);
}

package test2604;

/*
 * enum 이라는 기능이에요
 * */
enum Tri {
	// 요거 전체가 Tri 의 값이에요
    A("A"), B("AB"), C("ABC");
	// A 는 이름 "A" 가 실제값, C는 이름 "ABC"가 실제 값
	// 실제 값은 code 로 라는 놈으로 접근
    private String code;
 
    Tri(String code) {
        this.code = code;
    }
 
    public String code() {
        return code;
    }
}

public class Test2604 {
	public static void main(String[] args) {
		//  Tri.A.name() == A의 이름. 즉 A
		// Tri.values() == [A("A"), B("AB"), C("ABC")]
		Tri t = Tri.values()[Tri.A.name().length()];
        System.out.print(t.code());
	}
}
