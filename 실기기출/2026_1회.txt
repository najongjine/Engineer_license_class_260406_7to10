def f(a): # a= [1,2,3,4]
    # [[1], [2], [3], [4]]
    m = [[x] for x in a]
    
    b = m[:] # [[1], [2,1], [3,2,1], [4,3,2,1]]
    # i 가 0~2 까지 반복
    for i in range(len(b) - 1): # range(3)
        # b[2+1] += b[2] -> b[3]= [4] + [3,2,1]
        b[i+1] += b[i]
    print(f"forloop 이후의 b:",b)
    print(f"forloop 이후의 m:",m)
    # 길이([4,3,2,1])
    # sum(1, 2, 3, 4)
    return sum(len(x) for x in m)

print(f([1, 2, 3, 4]))
