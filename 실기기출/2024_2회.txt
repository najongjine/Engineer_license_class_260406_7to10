def fnCalculation(x,y):
    result = 0; # 1
    for i in range(len(x)): # 0 ~ 10, 2
     #      x[2:2+2]= dc
     temp = x[i:i+len(y)] 
     if temp == y: # dc == ab
       result += 1;
    return result
 
a = "abdcabcabca"

p2 = "ca";
#               ab3ca3

#include <stdio.h>
#include <string.h>
//  d = "teststring",  s = "first"
void sumFn(char* d, const char* s) {
    /*
    * d 라는 연속된 문자열 배열 공간에, s가 가지고 있던
    * 문자열을 덮어 써 버렸어요.
    * first\0ring
    */
    while (*s) {
        *d = *s;
        d++;
        s++;
    }
    *d = '\0';
}

int main() {
    const char* str1 = "first";
    char str2[50] = "teststring";
    int result = 0;
    sumFn(str2, str1);
    // i 0~4, 5 에서 빠져나가요
    for (int i = 0; str2[i] != '\0'; i++) {
        result += i;
    }
    printf("%d", result);

    return 0;
}
